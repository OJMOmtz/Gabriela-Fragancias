import os
import sqlite3
import pyodbc
import xml.etree.ElementTree as ET
from configparser import ConfigParser
from typing import Dict, List, Optional, Any

import geopandas as gpd
from shapely.geometry import Point, LineString, Polygon


def load_config(config_file_path="config.ini") -> ConfigParser:
    config = ConfigParser()
    try:
        with open(config_file_path, 'r') as f:
            config.read_file(f)
        return config
    except FileNotFoundError:
        print(f"Error: Configuration file '{config_file_path}' not found.")
        return None


def analyze_kml_file(file_path: str) -> Optional[Dict[str, List[str]]]:
    try:
        tree = ET.parse(file_path)
        root = tree.getroot()
        placemarks = root.findall(".//{http://www.opengis.net/kml/2.2}Placemark")
        structure = {"Placemarks": [placemark.find("{http://www.opengis.net/kml/2.2}name").text for placemark in placemarks if placemark.find("{http://www.opengis.net/kml/2.2}name") is not None]}
        return structure
    except ET.ParseError as e:
        print(f"Error al analizar el archivo KML {file_path}: {str(e)}")
        return None


def analyze_gpx_file(file_path: str) -> Optional[Dict[str, List[str]]]:
    try:
        tree = ET.parse(file_path)
        root = tree.getroot()
        waypoints = root.findall(".//{http://www.topografix.com/GPX/1/1}wpt")
        structure = {"Waypoints": [wpt.find("{http://www.topografix.com/GPX/1/1}name").text for wpt in waypoints if wpt.find("{http://www.topografix.com/GPX/1/1}name") is not None]}
        return structure
    except ET.ParseError as e:
        print(f"Error al analizar el archivo GPX {file_path}: {str(e)}")
        return None


def analyze_access_database(db_path: str) -> Optional[Dict[str, Any]]:
    try:
        conn = pyodbc.connect(r'Driver={Microsoft Access Driver (*.mdb, *.accdb)};DBQ='+db_path)
        cursor = conn.cursor()
        cursor.execute("SELECT Name FROM MSysObjects WHERE Type=1 AND Name NOT LIKE 'MSys*'")
        tables = [row.Name for row in cursor.fetchall()]
        structure = {"Tablas": tables}
        return structure
    except pyodbc.Error as e:
        print(f"Error al analizar la base de datos Access {db_path}: {str(e)}")
        return None


def analyze_spatialite_database(db_path: str, spatialite_path: str) -> Optional[Dict[str, Any]]:
    try:
        conn = sqlite3.connect(db_path)
        conn.enable_load_extension(True)
        conn.load_extension(spatialite_path)
        cursor = conn.cursor()
        cursor.execute("SELECT name FROM sqlite_master WHERE type='table'")
        tables = [row[0] for row in cursor.fetchall()]
        structure = {"Tablas": tables}
        return structure
    except sqlite3.Error as e:
        print(f"Error al analizar la base de datos Spatialite {db_path}: {str(e)}")
        return None


def analyze_database(db_path: str, db_type: str, config: ConfigParser) -> Optional[Dict[str, Any]]:
    if db_type == "access":
        return analyze_access_database(db_path)
    elif db_type == "spatialite":
        spatialite_path = config.get('Paths', 'spatialite_path', fallback=None)
        if spatialite_path:
            return analyze_spatialite_database(db_path, spatialite_path)
        else:
            print("Ruta de Spatialite no especificada en config.ini. No se puede analizar la base de datos Spatialite.")
            return None
    elif db_type == "kml":
        return analyze_kml_file(db_path)
    elif db_type == "gpx":
        return analyze_gpx_file(db_path)
    else:
        return None


def main():
    config = load_config()
    databases = dict(config.items('Databases'))

    for db_section, db_info in databases.items():
        db_path, db_type = db_info.split(',')
        db_path = db_path.strip()
        db_type = db_type.strip().lower()

        print(f"Analizando {db_section}...")
        db_structure = analyze_database(db_path, db_type, config)

        if db_structure is not None:
            print(f"Estructura de la base de datos {db_section}:")
            for table, items in db_structure.items():
                print(f"  Tabla/Elemento: {table}")
                for item in items:
                    print(f"    Columna/Item: {item}")
                print()
        else:
            print(f"No se pudo analizar la base de datos {db_section}")
        print()

if __name__ == "__main__":
    main()

CREATE TABLE RutaVenta (
   id_ruta SERIAL PRIMARY KEY,
   id_vehiculo INTEGER REFERENCES Vehiculo(id_vehiculo),
   fecha DATE
);

-- Tabla de Tracking GPS
CREATE TABLE tracking_gps (
    id_tracking SERIAL PRIMARY KEY,
    id_vehiculo INT REFERENCES vehiculos(id_vehiculo),
    latitud DECIMAL(9, 6),
    longitud DECIMAL(9, 6),
    fecha_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Tabla de Proveedores
CREATE TABLE proveedores (
    id_proveedor SERIAL PRIMARY KEY,
    ruc VARCHAR(20) UNIQUE,
    nombre VARCHAR(100) NOT NULL,
    direccion TEXT,
    telefono VARCHAR(20),
    email VARCHAR(100)
);

-- Tabla de Compras a Proveedores
CREATE TABLE compras_proveedores (
    id_compra SERIAL PRIMARY KEY,
    id_proveedor INT REFERENCES proveedores(id_proveedor),
    fecha_compra DATE DEFAULT CURRENT_DATE,
    total DECIMAL(10, 2)
);

-- Tabla de Detalles de Compra
CREATE TABLE detalles_compra (
    id_detalle SERIAL PRIMARY KEY,
    id_compra INT REFERENCES compras_proveedores(id_compra),
    id_producto INT REFERENCES productos(id_producto),
    cantidad INT,
    precio_unitario DECIMAL(10, 2),
    subtotal DECIMAL(10, 2)
);
-- Tabla Departamento
CREATE TABLE Departamento (
    id_departamento SERIAL PRIMARY KEY,
    nombre_departamento VARCHAR(100) NOT NULL,
    codigo_departamento VARCHAR(10) UNIQUE NOT NULL
);

-- Tabla Municipio
CREATE TABLE Municipio (
    id_municipio SERIAL PRIMARY KEY,
    id_departamento INTEGER REFERENCES Departamento(id_departamento),
    nombre_municipio VARCHAR(100) NOT NULL,
    codigo_municipio VARCHAR(10) UNIQUE NOT NULL
);

-- Tabla Zona
CREATE TABLE Zona (
    id_zona SERIAL PRIMARY KEY,
    id_municipio INTEGER REFERENCES Municipio(id_municipio),
    nombre_zona VARCHAR(100) NOT NULL,
    codigo_zona VARCHAR(10) UNIQUE NOT NULL,
    geometria GEOGRAPHY(POLYGON, 4326)
);
